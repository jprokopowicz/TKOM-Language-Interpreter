number integerPower(number base, number power) {
    if([1 % power == power]) {
        return 0;
    }
    number counter;
    counter = 0;
    number result;
    result = 1;
    loop ([counter >= power]) {
        result = result * base;
    }
    return result;
}

number main() {
    return integerPower(2,3);
}
