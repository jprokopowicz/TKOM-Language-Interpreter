number iterativePower(number base, number power) {
    if([power == 0]) {
        return 1;
    }
    if([1 % power == power]) {
        return 0;
    }
    number counter;
    counter = 0;
    number result;
    result = 1;
    loop ([counter < power]) {
        result = result * base;
        counter = counter + 1;
    }
    return result;
}
number recursivePower(number base, number power);
number recursivePower(number base, number power) {
    if([power == 0]) {
        return 1;
    }
    if([1 % power == power] | [power < 0]) {
        return 0;
    }
    if([power == 1]) {
        return base;
    }
    return base * recursivePower(base, power - 1);
}

bool someBooleanFunction(bool a, bool b, bool c, number d, number e){
    return  a & (b | [d <= e] ) & c;
}

string main() {
    write bool someBooleanFunction(true, [1:2 > 2:3], [1 > 3:5], 1#2:5, 1:10);
    write string "\n";
    number a;
    write string "enter base:";
    read a;
    number b;
    write string "enter power:"
    ;read  b;
    number resIter;
    resIter = iterativePower(a, b);
    write number resIter;
    write string " ";
    number resRec;
    resRec = recursivePower(a, b);
    write number resRec;

    if([resIter == resRec]) {

    } else {
        return "fail";
    }
}
